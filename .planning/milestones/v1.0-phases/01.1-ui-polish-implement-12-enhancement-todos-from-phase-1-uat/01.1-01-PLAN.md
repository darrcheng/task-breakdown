**TaskListItem.tsx — Two-click confirm for done->todo cycling (Todo #11):**

import { useState } from 'react';
```

2. Inside the `TaskListItem` component, add state for confirm-reopen:

const [confirmReopen, setConfirmReopen] = useState(false);
```

3. Replace the `handleStatusClick` function. When the task status is `done` and `confirmReopen` is false, enter confirm state for 2 seconds instead of cycling. When `confirmReopen` is true (second click), proceed with cycling:

```tsx
const handleStatusClick = async (e: React.MouseEvent) => {
  e.stopPropagation();
  if (task.id) {
    if (task.status === 'done' && !confirmReopen) {
      setConfirmReopen(true);
      setTimeout(() => setConfirmReopen(false), 2000);
      return;
    }
    setConfirmReopen(false);
    await db.tasks.update(task.id, {
      status: getNextStatus(task.status),
      updatedAt: new Date(),
    });
  }
};
```

4. Update the status circle button styling. When `confirmReopen` is true, show a pulsing ring indicator to signal "click again to reopen":

```tsx
<button
  onClick={handleStatusClick}
  className={clsx(
    'w-5 h-5 rounded-full border-2 flex-shrink-0 transition-colors',
    confirmReopen
      ? 'bg-emerald-500 border-emerald-500 ring-2 ring-offset-1 ring-amber-400 animate-pulse'
      : task.status === 'done'
        ? 'bg-emerald-500 border-emerald-500'
        : task.status === 'in-progress'
          ? 'bg-amber-400 border-amber-400'
          : 'bg-white border-slate-400 hover:border-slate-500'
  )}
  title={
    confirmReopen
      ? 'Click again to reopen task'
      : `Status: ${statusLabel}. Click to cycle.`
  }
/>
```

This matches the existing click-again-to-confirm pattern used for task/category deletion in the codebase (3s timeout for delete, 2s for status reopen since it's less destructive).
