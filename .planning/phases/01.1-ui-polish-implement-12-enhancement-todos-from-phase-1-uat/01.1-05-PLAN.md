**Create src/hooks/useSettings.ts (Todo #3):**

Create a custom hook that reads/writes settings to localStorage under key `'taskbreaker-settings'`.

```tsx
import { useState } from 'react';

const SETTINGS_KEY = 'taskbreaker-settings';

export interface Settings {
  weekStartsOn: 0 | 1;          // 0=Sunday, 1=Monday
  showKeyboardShortcuts: boolean; // toggle hint display in UI
}

const DEFAULT_SETTINGS: Settings = {
  weekStartsOn: 0,
  showKeyboardShortcuts: true,
};

export function useSettings() {
  const [settings, setSettings] = useState<Settings>(() => {
    try {
      const stored = localStorage.getItem(SETTINGS_KEY);
      return stored ? { ...DEFAULT_SETTINGS, ...JSON.parse(stored) } : DEFAULT_SETTINGS;
    } catch {
      return DEFAULT_SETTINGS;
    }
  });

  const updateSettings = (updates: Partial<Settings>) => {
    setSettings(prev => {
      const next = { ...prev, ...updates };
      localStorage.setItem(SETTINGS_KEY, JSON.stringify(next));
      return next;
    });
  };

  return { settings, updateSettings };
}
```

Export both the `Settings` interface and `useSettings` function.

3. Add keyboard shortcut handler (Todo #2). Use `useEffect` with proper guards:

```tsx
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    // Guard: don't fire in inputs
    const tag = (e.target as HTMLElement).tagName;
    if (['INPUT', 'TEXTAREA', 'SELECT'].includes(tag)) return;
    // Guard: don't fire with modifier keys
    if (e.metaKey || e.ctrlKey || e.altKey) return;
    // Guard: don't fire when any modal/overlay is open
    if (modalState.isOpen || isCategoryManagerOpen || isSettingsOpen) return;

    switch (e.key) {
      case 'j': // Next period
        setCurrentMonth(prev =>
          calendarView === 'week' ? addWeeks(prev, 1) : addMonths(prev, 1)
        );
        break;
      case 'k': // Previous period
        setCurrentMonth(prev =>
          calendarView === 'week' ? subWeeks(prev, 1) : subMonths(prev, 1)
        );
        break;
      case 't': // Today
        setCurrentMonth(new Date());
        break;
      case 'm': // Month view
        setCalendarView('month');
        setViewMode('calendar');
        break;
      case 'w': // Week view
        setCalendarView('week');
        setViewMode('calendar');
        break;
      case 'c': // Create task for today
        setModalState({ isOpen: true, date: formatDateKey(new Date()) });
        break;
      case '?': // Open settings
        setIsSettingsOpen(true);
        break;
    }
  };

  window.addEventListener('keydown', handleKeyDown);
  return () => window.removeEventListener('keydown', handleKeyDown);
}, [calendarView, viewMode, modalState.isOpen, isCategoryManagerOpen, isSettingsOpen]);
```
